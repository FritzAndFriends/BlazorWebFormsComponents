✅ After — Blazor + BWFC (.razor)
<ListView ID="productList"
    DataKeyNames="ProductID" GroupItemCount="4"
    TItem="Product" SelectMethod="GetProducts">
    <EmptyDataTemplate>
        <table>
            <tr><td>No data was returned.</td></tr>
        </table>
    </EmptyDataTemplate>
    <GroupTemplate>
        <tr id="itemPlaceholderContainer">
            <td id="itemPlaceholder"></td>
        </tr>
    </GroupTemplate>
    <ItemTemplate>
        <td>
            <table>
                <tr><td>
                  <a href="@($"/Product/{context.ProductName}")">
                    <image src='/Catalog/Images/Thumbs/@context.ImagePath'
                      width="100" height="75" border="1" />
                  </a>
                </td></tr>
                <tr><td>
                    <a href="@($"/Product/{context.ProductName}")">
                      @context.ProductName
                    </a><br />
                    <b>Price: </b>@context.UnitPrice.ToString("c")
                </td></tr>
            </table>
        </td>
    </ItemTemplate>
</ListView>
